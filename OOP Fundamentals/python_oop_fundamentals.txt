class DataProcessor:
    def __init__(self, data):
        """
        data: list of numbers
        """
        self.data = data

    def remove_none(self):
        self.data = [x for x in self.data if x is not None]

    def normalize(self):
        max_val = max(self.data)
        self.data = [x / max_val for x in self.data]

    def summary(self):
        return {
            "count": len(self.data),
            "min": min(self.data),
            "max": max(self.data),
            "avg": sum(self.data) / len(self.data)
        }

Usage
dp = DataProcessor([10, 20, None, 30])
dp.remove_none()
dp.normalize()
print(dp.summary())
